Example 5: Counting

        }

Example 7: Grouping logging

        var Countries = [{
            country: "India",
            cities: ["New Delhi", "Mumbai", "Chennai", "Patna", "Varanasi"]
        }, {
            country: "Nepal",
            cities: ["Kathmandu", "Pokhara", "Patan", "Biratnagar"]
        }];

        console.group("Countries");
        Countries.forEach(element => {
            console.group(element.country);
            // Nested Grouping
            element.cities.forEach(element => {
                console.log(element);
            });
            console.groupEnd();
        });
        console.groupEnd();

Example 10: Displaying an array of objects in table

        function Person(FirstName, LastName, Age) {
            this.FirstName = FirstName;
            this.LastName = LastName;
            this.Age = Age;
        }

        var amit = new Person("Amit", "Bajaj", 35);
        var john = new Person("John", "Smith", 28);
        var misha = new Person("Misha", "Kapoor", 13);

        console.table([amit, john, misha]);

        //Displaying only FirstName and LastName
        console.table([amit, john, misha], ["FirstName", "LastName"]);
